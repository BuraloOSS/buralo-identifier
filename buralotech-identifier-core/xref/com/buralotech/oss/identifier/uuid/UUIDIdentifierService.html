17  package com.buralotech.oss.identifier.uuid;
18  
19  import com.buralotech.oss.identifier.api.Identifier;
20  import com.buralotech.oss.identifier.api.IdentifierService;

22  import java.time.*;
23  import java.time.temporal.Temporal;
24  import java.util.Arrays;

26  /**
27   * Generate identifiers and parse binary and textual representations of identifiers. The generator uses either a Type 1
28   * UUID generator and juggles the bits so that the binary representations can be ordered by generation time or a new
29   * Type 6 UUID which is reordered in a similar way but is standardised. The textual representation is a modified
30   * URL-safe base 64 encoding that is also sortable.
31   */
32  public final class UUIDIdentifierService implements IdentifierService {
33  
34      /**
35       * Look-up table used during encoding.
36       */
37      private static final char[] ENCODING = {
38              '-', '0', '1', '2', '3', '4', '5', '6', '7', '8',
39              '9', 'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I',
40              'J', 'K', 'L', 'M', 'N', 'O', 'P', 'Q', 'R', 'S',
41              'T', 'U', 'V', 'W', 'X', 'Y', 'Z', '_', 'a', 'b',
42              'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l',
43              'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v',
44              'w', 'x', 'y', 'z'
45      };
46  
47      /**
48       * Look-up table used during decoding.
49       */
50      private static final int[] DECODING = new int[256];
51  
52      /*
53       * Initialise the array used for look-ups during decoding.
54       */
55      static {
56          // First mark all the values as invalid (-1)
57  
58          Arrays.fill(DECODING, -1);
59  
60          // Now map all the supported Base64 digits to their ordinal value
61  
62          for (int i = 0; i < ENCODING.length; i++) {
63              DECODING[ENCODING[i]] = i;
64          }
65      }
66  
67      /**
68       * Number of ticks per millisecond. The UUID tick is 100 nanoseconds.
69       */
70      private static final long TICKS_PER_MILLISECOND = 10000L;
71  
72      /**
73       * Number of ticks per second. TheUUID thick is 100 nanoseconds.
74       */
75      private static final long TICKS_PER_SECOND = 10000000L;
76  
77      /**
78       * The adjustment to apply to convert UUID epoch to Unix Epoch. The UUID epoch 15 October 1582, while the Unix
79       * epoch is 1 January 1970.
80       */
81      private static final long EPOCH_ADJ = 122192928000000000L;
82  
83      /**
84       * Delegate that encapsulates logic that is specific to the UUID format.
85       */
86      private final UUIDVersionDelegate delegate;
87  
88      /**
89       * Constructor used to inject the delegate that encapsulates the logic that is specific to the UUID format.
90       *
91       * @param delegate Encapsulates logic that is specific to the UUID format.
92       */
93      public UUIDIdentifierService(final UUIDVersionDelegate delegate) {
94          this.delegate = delegate;
95      }

97      /**
98       * Generate an identifier using an underlying UUID generator.
99       *
100      * @return The generated identifier.
101      */
102     @Override
103     public Identifier generate() {
104         final var binary = delegate.generate();
105         final var text = encode(binary);
106         return new UUIDIdentifier(text, binary);
107     }

109     /**
110      * Decode an identifier using its text representation.
111      *
112      * @param text The text representation.
113      * @return The identifier.
114      */
115     @Override
116     public Identifier fromText(final String text) {
117         if (text == null || !delegate.isValidText(text)) {
118             throw new IllegalArgumentException("invalid text representation of identifier");
119         }
120         final var binary = decode(text);
121         return new UUIDIdentifier(text, binary);
122     }
123 
124     /**
125      * Decode an identifier using its binary representation.
126      *
127      * @param binary The binary representation.
128      * @return The identifier.
129      */
130     @Override
131     public Identifier fromBinary(final byte[] binary) {
132         if (binary == null || binary.length != 16 || !delegate.isValidBinary(binary)) {
133             throw new IllegalArgumentException("invalid binary representation of identifier");
134         }
135         final var text = encode(binary);
136         return new UUIDIdentifier(text, binary);
137     }

139     /**
140      * Encode 16 bytes as 22 Base64 digit string.
141      *
142      * @param bytes The 16 input bytes.
143      * @return The 22 digit Base64 string.
144      */
145     private String encode(final byte[] bytes) {
146         assert bytes != null && bytes.length == 16;
147         final var chars = new char[22];
148         var i = 0;
149         var j = 0;
150         do {
151             encode3(bytes[i], bytes[i + 1], bytes[i + 2], chars, j);
152             i += 3;
153             j += 4;
154         } while (i < 15);
155         encode1(bytes[i], chars, j);
156         return new String(chars);
157     }
158 
159     /**
160      * Encode three bytes as 4 Base64 digits.
161      *
162      * @param b1    The first byte.
163      * @param b2    The second byte.
164      * @param b3    The third byte.
165      * @param chars The output character array in which the base 64 digits will be stored.
166      * @param j     The position at whichthe first Base64 digit will be stored.
167      */
168     private void encode3(final byte b1,
169                          final byte b2,
170                          final byte b3,
171                          final char[] chars,
172                          final int j) {
173         chars[j] = ENCODING[(b1 & 0xfc) >> 2];
174         chars[j + 1] = ENCODING[((b1 & 0x3) << 4) | ((b2 & 0xf0) >> 4)];
175         chars[j + 2] = ENCODING[((b2 & 0xf) << 2) | ((b3 & 0xc0) >> 6)];
176         chars[j + 3] = ENCODING[b3 & 0x3f];
177     }
178 
179     /**
180      * Encode a single byte as 2 Base64 digits.
181      *
182      * @param b1    The input byte.
183      * @param chars The output character array in which the base 64 digits will be stored.
184      * @param j     The position at whichthe first Base64 digit will be stored.
185      */
186     private void encode1(final byte b1,
187                          final char[] chars,
188                          final int j) {
189         chars[j] = ENCODING[(b1 & 0xfc) >> 2];
190         chars[j + 1] = ENCODING[(b1 & 0x3) << 4];
191     }
192 
193     /**
194      * Decode a 22 Base64 digit string into 16 bytes.
195      *
196      * @param str The Base64 digit string.
197      * @return The 16 bytes.
198      */
199     private byte[] decode(final String str) {
200         assert str != null && str.length() == 22;
201         final var bytes = new byte[16];
202         var i = 0;
203         var j = 0;
204         do {
205             decode4(bytes, i, str, j);
206             i += 3;
207             j += 4;
208         } while (i < 15);
209         decode2(bytes, i, str, j);
210         return bytes;
211     }
212 
213     /**
214      * Decode 4 Base64 digits into 3 bytes.
215      *
216      * @param bytes The destination where decoded bytes will be stored.
217      * @param i     The position at which to store the first decoded bytes.
218      * @param str   A string of Base64 digits.
219      * @param j     The position of the first Base64 digit.
220      */
221     private void decode4(final byte[] bytes,
222                          final int i,
223                          final String str,
224                          final int j) {
225         final var w = decode(str.charAt(j));
226         final var x = decode(str.charAt(j + 1));
227         final var y = decode(str.charAt(j + 2));
228         final var z = decode(str.charAt(j + 3));
229         bytes[i] = (byte) ((w << 2) | ((x & 0x30) >> 4));
230         bytes[i + 1] = (byte) (((x & 0xf) << 4) | ((y & 0x3c) >> 2));
231         bytes[i + 2] = (byte) (((y & 0x3) << 6) | z);
232     }
233 
234     /**
235      * Decode 2 Base64 digits into 1 byte.
236      *
237      * @param bytes  The destination where decoded bytes will be stored.
238      * @param dest   The position at which to store the decoded byte.
239      * @param string A string of Base64 digits.
240      * @param src    The position of the first Base64 digit.
241      */
242     private void decode2(final byte[] bytes,
243                          final int dest,
244                          final String string,
245                          final int src) {
246         final var w = decode(string.charAt(src));
247         final var x = decode(string.charAt(src + 1));
248         bytes[dest] = (byte) ((w << 2) | ((x & 0x30) >> 4));
249     }
250 
251     /**
252      * Decode a single Base64 digit.
253      *
254      * @param ch The Base64 digit.
255      * @return The value of the Base64 digit.
256      */
257     private int decode(final char ch) {
258         assert ch <= 256;
259         final var value = DECODING[ch];
260         assert value != -1;
261         return value;
262     }

264     /**
265      * Extract an instant from an identifier.
266      *
267      * @param identifier The identifier.
268      * @return The instant.
269      */
270     @Override
271     public Instant toInstant(final Identifier identifier) {
272         if (identifier == null) {
273             return null;
274         }
275         if (identifier instanceof UUIDIdentifier uuidIdentifier) {
276             return delegate.toInstant(uuidIdentifier.binary());
277 
278         } else {
279             throw new IllegalArgumentException("UUIDIdentifier is required");
280         }
281     }

283     /**
284      * Generate a lower-bound identifier for temporal value that can be used in range queries.
285      *
286      * @param time The temporal value ({@link java.time.Instant}, {@link java.time.LocalDate},
287      *             {@link java.time.LocalDateTime}, {@link java.time.OffsetDateTime},
288      *             {@link java.time.ZonedDateTime})
289      * @return A lower-bound identifier that can be used in a range query.
290      * @throws IllegalArgumentException If the temporal type is not supported.
291      */
292     @Override
293     public Identifier asLowerBound(final Temporal time) {
294         final var binary = delegate.fromTicks(toTicks(time, false), 0x8000000000000000L);
295         final var text = encode(binary);
296         return new UUIDIdentifier(text, binary);
297     }
298 
299     /**
300      * Generate an upper-bound identifier for temporal value that can be used in range queries.
301      *
302      * @param time The temporal value ({@link java.time.Instant}, {@link java.time.LocalDate},
303      *             {@link java.time.LocalDateTime}, {@link java.time.OffsetDateTime},
304      *             {@link java.time.ZonedDateTime})
305      * @return An upper-bound identifier that can be used in a range query.
306      * @throws IllegalArgumentException If the temporal type is not supported.
307      */
308     @Override
309     public Identifier asUpperBound(final Temporal time) {
310         final var binary = delegate.fromTicks(toTicks(time, true), 0x8FFFFFFFFFFFFFFFL);
311         final var text = encode(binary);
312         return new UUIDIdentifier(text, binary);
313     }

315     /**
316      * Given a temporal value extract the number of UUID ticks (100 nanoseconds).
317      *
318      * @param temporal The temporal value ({@link java.time.Instant}, {@link java.time.LocalDate},
319      *                 {@link java.time.LocalDateTime}, {@link java.time.OffsetDateTime},
320      *                 {@link java.time.ZonedDateTime})
321      * @return The number of ticks.
322      * @throws IllegalArgumentException If the temporal type is not supported.
323      */
324     private long toTicks(final Temporal temporal, final boolean upper) {
325         switch (temporal) {
326             case Instant instant -> {
327                 final var adj = EPOCH_ADJ + (upper ? TICKS_PER_MILLISECOND - 1 : 0L);
328                 return instant.toEpochMilli() * TICKS_PER_MILLISECOND + adj;
329             }
330             case LocalDate date -> {
331                 final var time = upper ? LocalTime.of(23, 59, 59) : LocalTime.of(0, 0, 0);
332                 final var adj = EPOCH_ADJ + (upper ? TICKS_PER_SECOND - 1 : 0L);
333                 return date.toEpochSecond(time, ZoneOffset.UTC) * TICKS_PER_SECOND + adj;
334             }
335             case LocalDateTime dateTime -> {
336                 final var adj = EPOCH_ADJ + (upper ? TICKS_PER_SECOND - 1 : 0L);
337                 return dateTime.toEpochSecond(ZoneOffset.UTC) * TICKS_PER_SECOND + adj;
338             }
339             case OffsetDateTime dateTime -> {
340                 final var adj = EPOCH_ADJ + (upper ? TICKS_PER_SECOND - 1 : 0L);
341                 return dateTime.toEpochSecond() * TICKS_PER_SECOND + adj;
342             }
343             case ZonedDateTime dateTime -> {
344                 final var adj = EPOCH_ADJ + (upper ? TICKS_PER_SECOND - 1 : 0L);
345                 return dateTime.toEpochSecond() * TICKS_PER_SECOND + adj;
346             }
347             default -> throw new IllegalArgumentException(temporal.getClass().getName() + " is not supported");
348         }
349     }
