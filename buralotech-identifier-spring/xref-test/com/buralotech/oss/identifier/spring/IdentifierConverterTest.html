17  package com.buralotech.oss.identifier.spring;
18  
19  import com.buralotech.oss.identifier.uuid.UUIDIdentifier;
20  import com.buralotech.oss.identifier.uuid.UUIDIdentifierService;
21  import com.buralotech.oss.identifier.uuid.UUIDVersion1Delegate;
22  import org.junit.jupiter.params.ParameterizedTest;
23  import org.junit.jupiter.params.provider.Arguments;
24  import org.junit.jupiter.params.provider.MethodSource;
25  import org.junit.jupiter.params.provider.NullSource;
26  import org.junit.jupiter.params.provider.ValueSource;
27  
28  import java.util.stream.Stream;
29  
30  import static org.assertj.core.api.Assertions.assertThat;
31  import static org.assertj.core.api.Assertions.assertThatThrownBy;
32  import static org.junit.jupiter.params.provider.Arguments.arguments;
33  
34  class IdentifierConverterTest {
35  
36      private final IdentifierConverter identifierConverter = new IdentifierConverter(new UUIDIdentifierService(new UUIDVersion1Delegate()));
37  
38      static Stream<Arguments> conversionSuccesses() {
39          return Stream.of(
40                  arguments("3TnesPWMmyqTHniq6DPq0F", new byte[]{17, -20, -22, -31, -88, 87, -53, -19, -98, 75, 59, -74, 28, -26, -74, 5}),
41                  arguments("3TnesPWMmyuTHniq6DPq0F", new byte[]{17, -20, -22, -31, -88, 87, -53, -18, -98, 75, 59, -74, 28, -26, -74, 5})
42          );
43      }
44  
45  
46      @ParameterizedTest
47      @MethodSource
48      void conversionSuccesses(final String text, final byte[] binary) {
49          final var identifier = new UUIDIdentifier(text, binary);
50          assertThat(identifierConverter.convert(text)).isEqualTo(identifier);
51      }
52  
53      @ParameterizedTest
54      @NullSource
55      @ValueSource(strings = {
56              "",
57              " ",
58              "3TnesPWMmyqTHniq6DPq0",
59              "3TnesPWMmyuTHniq6DPq0J",
60              "ITnesPWMmyuTHniq6DPq0F",
61              "3TnesPWMmyuTHniq6DPq0Fk"
62      })
63      void conversionFailures(final String text) {
64          assertThatThrownBy(() -> identifierConverter.convert(text))
65                  .isInstanceOf(IllegalArgumentException.class);
66      }
67  }
